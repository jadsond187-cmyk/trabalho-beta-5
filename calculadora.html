def soma(a, b): return a + b def subtracao(a, b): return a - b def multiplicacao(a, b): return a * b def divisao(a, b): if b == 0: return "Erro: divisão por zero" return a / b print("=== CALCULADORA ===") print("1 - Soma") print("2 - Subtração") print("3 - Multiplicação") print("4 - Divisão") opcao = int(input("Escolha uma operação: ")) num1 = float(input("Digite o primeiro número: ")) num2 = float(input("Digite o segundo número: ")) if opcao == 1: resultado = soma(num1, num2) elif opcao == 2: resultado = subtracao(num1, num2) elif opcao == 3: resultado = multiplicacao(num1, num2) elif opcao == 4: resultado = divisao(num1, num2) else: resultado = "Opção inválida" print("Resultado:", resultado)
